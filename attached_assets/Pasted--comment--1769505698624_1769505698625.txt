{%- comment -%}
----------------------------------------------------------------------------------------------------------------------
VARIANT PICKER
----------------------------------------------------------------------------------------------------------------------

Render the variant picker option selectors.

********************************************
Supported variables
********************************************

* product: the product for which to render the selector (required)
* form_id: the ID of the form to which the variant picker belongs (required)
* update_url: if set to true, the URL is updated when the variant changes (usually used for product page)
* hide_sold_out_variants: if set to true, the sold out or unavailable variants are hidden
* hide_size_chart: if set to true, size chart is always hidden
* force_dropdown_as_block: if set to true, the dropdown selectors are replaced by blocks, which is useful for quick buy where dropdown are not convenient to use
* block: the block itself, for theme-specific options (such as selector types)
{%- endcomment -%}

{%- assign size_label_list = 'general.label.size' | t | replace: ', ', ',' | downcase | split: ',' -%}
{%- assign variant_image_options = block.settings.variant_image_options | replace: ', ', ',' | downcase | split: ',' -%}

          assign resolved_option_selector_style = 'variant_image'
        endif

      <script>
        (function() {
          const variantPicker = document.currentScript?.closest('variant-picker');
          if (!variantPicker) return;
          
          const form = variantPicker.closest('form') || document.querySelector(`form#${variantPicker.getAttribute('form-id')}`);
          if (!form) return;
          
          // Function to find Frame and Mount option positions
          function findOptionPositions() {
            const optionInputs = form.querySelectorAll('input[data-option-position]');
            let framePosition = null;
            let mountPosition = null;
            
            optionInputs.forEach(input => {
              const label = input.closest('fieldset')?.querySelector('legend');
              if (label) {
                const optionName = label.textContent.replace(':', '').trim().toLowerCase();
                const position = parseInt(input.getAttribute('data-option-position'));
                
                if (optionName === 'frame') {
                  framePosition = position;
                } else if (optionName === 'mount') {
                  mountPosition = position;
                }
              }
            });
            
            return { framePosition, mountPosition };
          }
          
          // Function to automatically set Mount to "No" when Unframed is selected
          function handleFrameChange() {
            const { framePosition, mountPosition } = findOptionPositions();
            if (!framePosition || !mountPosition) return;
            
            // Find the selected Frame option
            const frameInputs = form.querySelectorAll(`input[data-option-position="${framePosition}"]`);
            const selectedFrameInput = Array.from(frameInputs).find(input => input.checked);
            
            if (!selectedFrameInput) return;
            
            // Get the label text for the selected frame option - check multiple ways
            const frameLabel = selectedFrameInput.closest('label, .frame-option-box');
            if (!frameLabel) return;
            
            // Get text from label or from the input's value/ID
            let frameText = frameLabel.textContent.trim().toLowerCase();
