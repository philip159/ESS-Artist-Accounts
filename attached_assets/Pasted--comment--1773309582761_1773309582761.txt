{%- comment -%}
----------------------------------------------------------------------------------------------------------------------
VARIANT PICKER
----------------------------------------------------------------------------------------------------------------------

Render the variant picker option selectors.

********************************************
Supported variables
********************************************

* product: the product for which to render the selector (required)
* form_id: the ID of the form to which the variant picker belongs (required)
* update_url: if set to true, the URL is updated when the variant changes (usually used for product page)
* hide_sold_out_variants: if set to true, the sold out or unavailable variants are hidden
* hide_size_chart: if set to true, size chart is always hidden
* force_dropdown_as_block: if set to true, the dropdown selectors are replaced by blocks, which is useful for quick buy where dropdown are not convenient to use
* block: the block itself, for theme-specific options (such as selector types)
{%- endcomment -%}

{%- assign color_label_list = 'general.label.color' | t | replace: ', ', ',' | downcase | split: ',' -%}
{%- assign size_label_list = 'general.label.size' | t | replace: ', ', ',' | downcase | split: ',' -%}
{%- assign variant_image_options = block.settings.variant_image_options | replace: ', ', ',' | downcase | split: ',' -%}

{%- comment -%}Debug: product.has_only_default_variant = {{ product.has_only_default_variant }}, product.options_with_values.size = {{ product.options_with_values.size }}{%- endcomment -%}
{%- unless product.has_only_default_variant -%}
  <variant-picker class="variant-picker" section-id="{{ section.id }}" handle="{{ product.handle }}" form-id="{{ form_id }}" {% if update_url %}update-url{% endif %}>
    {%- comment -%}
    The variant data is outputted as a JSON, which allows the theme to emit an event with the data when the variant changes. This must not be removed.
    {%- endcomment -%}
    <script data-variant type="application/json">
      {{- product.selected_or_first_available_variant | json -}}
    </script>

    {%- for option in product.options_with_values -%}
      {% liquid
        assign option_downcase = option.name | downcase
        assign next_index = forloop.index0 | plus: 1
        assign next_option = product.options_with_values[next_index]
        assign next_option_downcase = next_option.name | downcase
        assign resolved_option_selector_style = block.settings.selector_style
        assign is_frame_option = false
        assign is_mount_option = false
        
        # Force Frame option to use block style
        if option_downcase == 'frame'
          assign resolved_option_selector_style = 'block'
          assign is_frame_option = true
        endif
        
        # Force Mount option to use block style (split bar)
        if option_downcase == 'mount'
          assign resolved_option_selector_style = 'block'
          assign is_mount_option = true
        endif

        # Mount variants are now handled by the add-on widget, so skip rendering them in the picker.
        if is_mount_option
          continue
        endif

        assign swatch_count = option.values | map: 'swatch' | compact | size

        if swatch_count > 0 and block.settings.swatch_selector_style != 'none' and is_frame_option == false and is_mount_option == false
          # Use the swatch selector type only if we have at least one swatch and a supported swatch selector type
          assign resolved_option_selector_style = block.settings.swatch_selector_style
        endif

        # Implementation note: if the option value has no native swatch, BUT that the option name matches a hardcoded list of color names,
        # we fallback to the legacy config-based system. This allows to keep compatibility with stores that were using the config-based, and
        # allow those merchants to upgrade to the new system at their own pace.
        if swatch_count == 0 and color_label_list contains option_downcase and block.settings.swatch_selector_style != 'none' and is_frame_option == false and is_mount_option == false
          assign resolved_option_selector_style = block.settings.swatch_selector_style
        endif

        if resolved_option_selector_style == 'dropdown' and force_dropdown_as_block
          assign resolved_option_selector_style = 'block'
        endif

        if variant_image_options contains option_downcase and is_frame_option == false and is_mount_option == false
          assign resolved_option_selector_style = 'variant_image'
        endif
        
        # Ensure mount and frame options always use block style
        if is_mount_option or is_frame_option
          assign resolved_option_selector_style = 'block'
        endif
      %}

      <fieldset class="variant-picker__option {% if is_frame_option %}variant-picker__option--frame{% endif %} {% if is_mount_option %}variant-picker__option--mount{% endif %}" data-option-name="{{ option.name | downcase }}" data-option-position="{{ option.position }}">
        <div class="variant-picker__option-info">
          <div class="h-stack gap-2">
            <legend class="{% if is_frame_option %}variant-picker__option-label--frame{% endif %} {% if is_mount_option %}variant-picker__option-label--mount{% endif %}">{{ option.name }}:</legend>
            <span class="{% if is_frame_option %}variant-picker__option-selected--frame{% endif %} {% if is_mount_option %}variant-picker__option-selected--mount{% endif %}">{{ option.selected_value }}</span>
          </div>

          {%- if hide_size_chart != true and block.settings.size_chart_page != blank and size_label_list contains option_downcase -%}
            {%- capture drawer_id -%}size-chart-{{ option.position }}-{{ form_id }}{%- endcapture -%}

            <button type="button" class="text-sm text-subdued" aria-controls="{{ drawer_id | escape }}" aria-expanded="false">
              <span class="link">{{ 'product.general.size_chart' | t }}</span>
            </button>

            <x-drawer id="{{ drawer_id }}" class="drawer drawer--lg">
              <span class="h5" slot="header">{{ block.settings.size_chart_page.title }}</span>

              <div class="prose">
                {{- block.settings.size_chart_page.content -}}
              </div>
            </x-drawer>
          {%- endif -%}
        </div>

        {%- if resolved_option_selector_style == 'dropdown' -%}
          {%- capture popover_id -%}popover-variant-picker-{{ section.id }}-{{ product.id }}-{{ option.position }}{%- endcapture -%}

          <div class="relative">
            <button type="button" class="select" aria-controls="{{ popover_id }}" aria-expanded="false">
              <span id="{{ popover_id }}-selected-value">{{- option.selected_value -}}</span>
              {%- render 'icon' with 'chevron-bottom', class: 'select-chevron' -%}
            </button>

            <x-popover id="{{ popover_id }}" class="popover" initial-focus="[aria-selected='true']" close-on-listbox-select anchor-horizontal="start" anchor-vertical="end">
              <p class="h5" slot="title">{{ option.name }}</p>

              {%- assign param_name = form_id | append: '-option' | append: option.position -%}

              <div data-option-selector class="popover-listbox">
                {%- for option_value in option.values -%}
                  {%- if hide_sold_out_variants == false or option_value.available or option_value.selected -%}
                    {%- if update_url and option_value.product_url != blank -%}
                      {%- if option_value.selected -%}
                        <input class="sr-only" type="radio" id="{{ param_name }}-{{ option_value.id }}" form="{{ form_id }}" name="{{ param_name }}" data-option-position="{{ option.position }}" value="{{ option_value.id }}" checked>
                      {%- endif -%}

                      <a href="{{ option_value.variant.url | default: option_value.product_url }}" class="popover-listbox__option {% unless option_value.available %}is-disabled{% endunless %}">
                        {{- option_value.name -}}
                      </a>
                    {%- else -%}
                      <label class="popover-listbox__option {% unless option_value.available %}is-disabled{% endunless %}" for="{{ param_name }}-{{ option_value.id }}">
                        <input class="sr-only" form="{{ form_id }}" type="radio" id="{{ param_name }}-{{ option_value.id }}" name="{{ param_name }}" {% if option_value.product_url != blank %}data-product-url="{{ option_value.product_url | escape }}"{% endif %} data-option-position="{{ option.position }}" value="{{ option_value.id }}" {% if option_value.selected %}checked{% endif %}>

                        {{- option_value.name -}}
                      </label>
                    {%- endif -%}
                  {%- endif -%}
                {%- endfor -%}
              </div>
            </x-popover>
          </div>
        {%- else -%}
          <div {% unless block.settings.stack_blocks %}class="scroll-area bleed sm:unbleed"{% endunless %}>
            <div class="variant-picker__option-values {% if is_frame_option %}variant-picker__option-values--frame{% endif %} {% if is_mount_option %}variant-picker__option-values--mount{% endif %} {% if block.settings.stack_blocks %}wrap{% else %}scroll-area bleed sm:unbleed{% endif %} {% if resolved_option_selector_style == 'swatch' and settings.color_swatch_style == 'rectangle' %}variant-picker__option-values--color gap-4{% else %}gap-2{% endif %}">
              {%- if is_mount_option -%}
                {%- comment -%}Custom Mount option split bar{%- endcomment -%}
                {%- assign name = form_id | append: '-option' | append: option.position -%}
                {%- for option_value in option.values -%}
                  {%- if hide_sold_out_variants == false or option_value.available or option_value.selected -%}
                    {%- liquid
                      assign mount_box_id = name | append: '-' | append: option_value.id
                      assign selected = option_value.selected
                      assign disabled = false
                      if option_value.available == false
                        assign disabled = true
                      endif
                    -%}
                    
                    <label class="mount-option-box {% if selected %}is-selected{% endif %} {% if disabled %}is-disabled{% endif %}" for="{{ mount_box_id }}">
                      <input class="sr-only" form="{{ form_id }}" type="radio" id="{{ mount_box_id }}" name="{{ name }}" data-option-position="{{ option.position }}" value="{{ option_value.id }}" {% if selected %}checked{% endif %}>
                      <span class="mount-option-box__label">{{ option_value.name }}</span>
                    </label>
                  {%- endif -%}
                {%- endfor -%}
              {%- elsif is_frame_option -%}
                {%- comment -%}Custom Frame option boxes with images{%- endcomment -%}
                {%- assign name = form_id | append: '-option' | append: option.position -%}
                {%- for option_value in option.values -%}
                  {%- if hide_sold_out_variants == false or option_value.available or option_value.selected -%}
                    {%- liquid
                      assign option_value_downcase = option_value.name | downcase
                      assign frame_image = null
                      assign frame_image_position = 'center'
                      
                      # Get variant name lists and check for matches
                      # Use custom names if provided, otherwise use defaults
                      if block.settings.frame_variant_names_black != blank
                        assign black_names = block.settings.frame_variant_names_black | downcase | split: ','
                      else
                        assign black_names = 'black' | split: ','
                      endif
                      
                      if block.settings.frame_variant_names_white != blank
                        assign white_names = block.settings.frame_variant_names_white | downcase | split: ','
                      else
                        assign white_names = 'white' | split: ','
                      endif
                      
                      if block.settings.frame_variant_names_natural != blank
                        assign natural_names = block.settings.frame_variant_names_natural | downcase | split: ','
                      else
                        assign natural_names = 'natural' | split: ','
                      endif
                      
                      if block.settings.frame_variant_names_unframed != blank
                        assign unframed_names = block.settings.frame_variant_names_unframed | downcase | split: ','
                      else
                        assign unframed_names = 'unframed' | split: ','
                      endif
                      
                      # Check each list for a match
                      assign matched = false
                      
                      for name in black_names
                        assign name_trimmed = name | strip
                        if option_value_downcase contains name_trimmed
                          assign frame_image = block.settings.frame_image_black
                          assign frame_image_position = block.settings.frame_image_black_position | default: 'center'
                          assign matched = true
                          break
                        endif
                      endfor
                      
                      unless matched
                        for name in white_names
                          assign name_trimmed = name | strip
                          if option_value_downcase contains name_trimmed
                            assign frame_image = block.settings.frame_image_white
                            assign frame_image_position = block.settings.frame_image_white_position | default: 'center'
                            assign matched = true
                            break
                          endif
                        endfor
                      endunless
                      
                      unless matched
                        for name in natural_names
                          assign name_trimmed = name | strip
                          if option_value_downcase contains name_trimmed
                            assign frame_image = block.settings.frame_image_natural
                            assign frame_image_position = block.settings.frame_image_natural_position | default: 'center'
                            assign matched = true
                            break
                          endif
                        endfor
                      endunless
                      
                      unless matched
                        for name in unframed_names
                          assign name_trimmed = name | strip
                          if option_value_downcase contains name_trimmed
                            assign frame_image = block.settings.frame_image_unframed
                            assign frame_image_position = block.settings.frame_image_unframed_position | default: 'center'
                            assign matched = true
                            break
                          endif
                        endfor
                      endunless
                      
                      # Set up frame box variables
                      assign frame_box_id = name | append: '-' | append: option_value.id
                      assign selected = option_value.selected
                      assign disabled = false
                      if option_value.available == false
                        assign disabled = true
                      endif
                    -%}
                    
                    {%- liquid
                      # Shopify checkboxes return "true" as a string when checked
                      assign hide_labels = false
                      if block.settings.frame_hide_labels == true or block.settings.frame_hide_labels == 'true'
                        assign hide_labels = true
                      endif
                      
                      assign image_crop = block.settings.frame_image_crop | default: 'cover'
                      assign crop_class = 'object-cover'
                      if image_crop == 'contain'
                        assign crop_class = 'object-contain'
                      elsif image_crop == 'fill'
                        assign crop_class = 'object-fill'
                      endif
                      
                      # Map position to object-position CSS values
                      assign object_position = 'center'
                      case frame_image_position
                        when 'top'
                          assign object_position = 'center top'
                        when 'bottom'
                          assign object_position = 'center bottom'
                        when 'left'
                          assign object_position = 'left center'
                        when 'right'
                          assign object_position = 'right center'
                        when 'top_left'
                          assign object_position = 'left top'
                        when 'top_right'
                          assign object_position = 'right top'
                        when 'bottom_left'
                          assign object_position = 'left bottom'
                        when 'bottom_right'
                          assign object_position = 'right bottom'
                        else
                          assign object_position = 'center'
                      endcase
                    -%}
                    <label class="frame-option-box {% if selected %}is-selected{% endif %} {% if disabled %}is-disabled{% endif %} {% if hide_labels %}frame-option-box--no-label{% endif %}" for="{{ frame_box_id }}">
                      <input class="sr-only" form="{{ form_id }}" type="radio" id="{{ frame_box_id }}" name="{{ name }}" data-option-position="{{ option.position }}" value="{{ option_value.id }}" {% if selected %}checked{% endif %}>
                      {%- if frame_image != blank -%}
                        <div class="frame-option-box__image" style="--image-position: {{ object_position }};">
                          {{- frame_image | image_url: width: 200 | image_tag: loading: 'lazy', sizes: '120px', widths: '120,200', class: crop_class, style: 'object-position: var(--image-position);' -}}
                        </div>
                      {%- endif -%}
                      {%- if hide_labels == false -%}
                        <span class="frame-option-box__label">{{ option_value.name }}</span>
                      {%- endif -%}
                    </label>
                  {%- endif -%}
                {%- endfor -%}
              {%- else -%}
                {%- comment -%}Standard option rendering for non-frame options{%- endcomment -%}
                {% liquid
                  assign name = form_id | append: '-option' | append: option.position

                  for option_value in option.values
                    case resolved_option_selector_style
                      when 'variant_image'
                        render 'option-value', type: 'thumbnail', form: form_id, option_value: option_value, param_name: name, option_position: option.position, hide_if_disabled: hide_sold_out_variants, reload_page_for_combined_products: update_url, id_prefix: forloop.index, bordered: true
                      when 'swatch'
                        render 'option-value', type: 'swatch', form: form_id, option_value: option_value, param_name: name, option_position: option.position, hide_if_disabled: hide_sold_out_variants, reload_page_for_combined_products: update_url, id_prefix: forloop.index
                      when 'block'
                        render 'option-value', type: 'block', form: form_id, option_value: option_value, param_name: name, option_position: option.position, hide_if_disabled: hide_sold_out_variants, reload_page_for_combined_products: update_url, id_prefix: forloop.index
                      when 'block_swatch'
                        render 'option-value', type: 'block', form: form_id, option_value: option_value, param_name: name, option_position: option.position, show_swatch: true, hide_if_disabled: hide_sold_out_variants, reload_page_for_combined_products: update_url, id_prefix: forloop.index
                    endcase
                  endfor
                %}
              {%- endif -%}
            </div>
          </div>
        {%- endif -%}
      </fieldset>

    {%- endfor -%}
  </variant-picker>

      <script>
        (function () {
          try {
            const FORM_ID = '{{ form_id }}';
            const variantPicker = document.querySelector(`variant-picker[form-id="${FORM_ID}"]`);
            if (!variantPicker) return;

            const form = document.getElementById(FORM_ID) || variantPicker.closest('form');
            if (!form) return;

            let scheduled = false;

            const isVisible = (element) => {
              return element && element.getClientRects().length > 0;
            };

            const getAvailableFrameInputs = () => {
              const frameFieldset = variantPicker.querySelector('fieldset[data-option-name="frame"]');
              if (!frameFieldset) return [];

              return Array.from(
                frameFieldset.querySelectorAll('input[type="radio"][data-option-position]')
              ).filter((input) => {
                if (input.disabled) return false;
                if (input.closest('.is-disabled')) return false;

                const label = input.closest('label');
                return isVisible(label || input);
              });
            };

            const enforceSingleFrameSelection = () => {
              const frameInputs = getAvailableFrameInputs();
              if (frameInputs.length !== 1) return;

              const onlyInput = frameInputs[0];
              if (onlyInput.checked) return;

              const onlyLabel = onlyInput.closest('label');

              // Use click first so the theme's variant system handles all side effects.
              if (onlyLabel && !onlyInput.disabled) {
                onlyLabel.click();
                return;
              }

              onlyInput.checked = true;
              onlyInput.dispatchEvent(new Event('input', { bubbles: true }));
              onlyInput.dispatchEvent(new Event('change', { bubbles: true }));
            };

            const scheduleEnforce = () => {
              if (scheduled) return;
              scheduled = true;

              requestAnimationFrame(() => {
                scheduled = false;
                enforceSingleFrameSelection();
              });
            };

            form.addEventListener('change', scheduleEnforce, true);
            variantPicker.addEventListener('product:variant-change', scheduleEnforce);

            // Run once on load and once shortly after initial render settles.
            scheduleEnforce();
            setTimeout(enforceSingleFrameSelection, 60);
          } catch (error) {
            console.error('[Frame Auto-Select] error', error);
          }
        })();
      </script>
